-- build
CREATE TABLE Object(
	oid int NOT NULL AUTO_INCREMENT,
    item varchar(70) NOT NULL,
    size varchar(20),
    color varchar(50),
    unit varchar(30),
    ret int NOT NULL,
    com varchar(50),
    PRIMARY KEY(oid)
);

CREATE TABLE Place(
	pid int NOT NULL AUTO_INCREMENT,
    name varchar(50),
    PRIMARY KEY(pid)
);

CREATE TABLE Category(
	catid int NOT NULL AUTO_INCREMENT,
	name varchar(80),
    parent int,
    PRIMARY KEY(catid)
);

CREATE TABLE Userinfo(
	uid int NOT NULL AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    acnt varchar(50) NOT NULL,
    pwd varchar(50) NOT NULL,
    PRIMARY KEY(uid)
);

CREATE TABLE Role(
	rid int NOT NULL AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    PRIMARY KEY(rid)
);

CREATE TABLE Actions(
	aid int NOT NULL AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    PRIMARY KEY(aid)
);

CREATE TABLE Semester(
	sid int NOT NULL AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    PRIMARY KEY(sid)
);

CREATE TABLE Class(
    cid int NOT NULL AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    PRIMARY KEY(cid)
);

CREATE TABLE Activity(
	eid int NOT NULL AUTO_INCREMENT,
    name varchar(80) NOT NULL,
    PRIMARY KEY(eid)
);

CREATE TABLE inAndOut(
	ioid int NOT NULL AUTO_INCREMENT,
    sid int NOT NULL,
    uid int NOT NULL,
    aid int NOT NULL,
    oid int NOT NULL,
    eid int NOT NULL,
    num float NOT NULL,
    get int NOT NULL,
    ret int NOT NULL,
    review int NOT NULL,
    dt date,
    reason varchar(150),
    PRIMARY KEY(ioid),
    FOREIGN KEY(sid) REFERENCES Semester(sid),
    FOREIGN KEY(uid) REFERENCES Userinfo(uid),
    FOREIGN KEY(aid) REFERENCES Actions(aid),
    FOREIGN KEY(oid) REFERENCES Object(oid),
    FOREIGN KEY(eid) REFERENCES Activity(eid)
);

CREATE TABLE userAndRole(
	urid int NOT NULL AUTO_INCREMENT,
    uid int NOT NULL,
    rid int NOT NULL,
    PRIMARY KEY(urid),
    FOREIGN KEY(uid) REFERENCES Userinfo(uid),
    FOREIGN KEY(rid) REFERENCES Role(rid)
);

CREATE TABLE userAndClass(
    ucid int NOT NULL AUTO_INCREMENT,
    uid int NOT NULL,
    cid int NOT NULL,
    PRIMARY KEY(ucid),
    FOREIGN KEY(uid) REFERENCES Userinfo(uid),
    FOREIGN KEY(cid) REFERENCES Class(cid)
);

CREATE TABLE OLimit(
	lid int NOT NULL AUTO_INCREMENT,
    oid int NOT NULL,
    num int NOT NULL,
    PRIMARY KEY(lid),
    FOREIGN KEY(oid) REFERENCES Object(oid)
);

CREATE TABLE SLimit(
    slid int NOT NULL AUTO_INCREMENT,
    uid int NOT NULL,
    oid int NOT NULL,
    num int NOT NULL,
    PRIMARY KEY(slid),
    FOREIGN KEY(uid) REFERENCES Userinfo(uid),
    FOREIGN KEY(oid) REFERENCES Object(oid)
);

CREATE TABLE Inventory(
	inid int NOT NULL AUTO_INCREMENT,
    oid int NOT NULL,
    num float NOT NULL,
    PRIMARY KEY(inid),
    FOREIGN KEY(oid) REFERENCES Object(oid)
);

CREATE TABLE objectAndCat(
	ocid int NOT NULL AUTO_INCREMENT,
    oid int NOT NULL,
    catid int NOT NULL,
    PRIMARY KEY(ocid),
    FOREIGN KEY(oid) REFERENCES Object(oid),
    FOREIGN KEY(catid) REFERENCES Category(catid)
);

CREATE TABLE objectAndPlace(
	opid int NOT NULL AUTO_INCREMENT,
    oid int NOT NULL,
    pid int NOT NULL,
    PRIMARY KEY(opid),
    FOREIGN KEY(oid) REFERENCES Object(oid),
    FOREIGN KEY(pid) REFERENCES Place(pid)
);
